实验9_1_括号匹配
题目描述
检测字符串中的圆括号 ( 和 ) 是否配对。

输入格式
一个长度不超过100的非空字符串，不含空格。

输出格式
匹配输出 parentheses match!，否则 parentheses do not match!

样例输入一
text
((a+b)(c+d))
样例输出一
text
parentheses match!
样例输入二
text
((a+b))c+d))
样例输出二
text
parentheses do not match!


#include<stdio.h>
#include<string.h>
int main(){
    char p[100];
    int len=0;
    gets(p);
    for(int i=0;p[i]!='\0';i++){
        if(p[i]=='('){
            len++;
        }
        if(p[i]==')'){
            len--;
        }
        if(len<0){
            break;
        }
    }
    if(len==0){
        printf("parentheses match!");
    }
    else{
        printf("parentheses do not match!");
    }
    return 0;
}

#include<stdio.h>

int main() {
    char a[101], b[101];  // b 作为栈使用
    scanf("%s", a);
    
    int k = 0;  // 栈顶指针
    
    for(int i = 0; a[i] != '\0'; i++) {
        // 左括号入栈
        if(a[i] == '{' || a[i] == '[' || a[i] == '(') {
            b[k++] = a[i];
        }
        
        // 右括号处理
        else if(a[i] == '}') {
            if(k > 0 && b[k-1] == '{') {
                k--;  // 匹配成功，出栈
            } else {
                printf("parentheses do not match!");
                return 0;
            }
        }
        else if(a[i] == ']') {
            if(k > 0 && b[k-1] == '[') {
                k--;
            } else {
                printf("parentheses do not match!");
                return 0;
            }
        }
        else if(a[i] == ')') {
            if(k > 0 && b[k-1] == '(') {
                k--;
            } else {
                printf("parentheses do not match!");
                return 0;
            }
        }
        // 其他字符忽略
    }
    
    // 循环结束后，栈应为空
    if(k == 0) {
        printf("parentheses match!");
    } else {
        printf("parentheses do not match!");
    }
    
    return 0;
}
