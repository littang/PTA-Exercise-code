简单点的方法
void InsertSort(int a[], int n){
    for(int i=1; i<n; i++){
        int key = a[i];
        int j = i-1;
        while(j>=0 && a[j] > key){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = key;
        // 输出当前状态
        printf("%d", a[0]);
        for(int k=1; k<n; k++) printf(" %d", a[k]);
        printf("\n");
    }
}

void InsertSort(int a[], int n){
  for(int i=1;i<n;i++){
    for(int j=i-1;j>=0;j--){
      for(int k=0;j-k>=0;k++)
        if(a[i-k]<a[j-k]){
          int swap=a[i-k];
          a[i-k]=a[j-k];
          a[j-k]=swap;
        }
      
    }
    printarray(a,n);
  }
}
