选择排序
题目描述
实现选择排序，输出每一趟排序后的结果。

void selectSort(int data[], int elementCount);
int findMin(int data[], int startLoc, int endLoc);
void outputData(int data[], int elementCount);
裁判测试程序样例
c
#include<stdio.h>
void selectSort(int data[],int elementCount);
int findMin(int data[], int startLoc, int endLoc);
void outputData(int data[],int elementCount);
int main()
{
    int n , i, num[1010] ;
    scanf("%d",&n); 
    for( i = 0 ; i < n ; i++ ) scanf("%d",&num[i]) ;
    selectSort(num,n) ; 
    return 0 ;
}

int findMin(int data[], int startLoc, int endLoc){
    int minIdx = startLoc;
    for(int i=startLoc+1; i<=endLoc; i++){
        if(data[i] < data[minIdx]) minIdx = i;
    }
    return minIdx;
}
void outputData(int data[], int elementCount){
    printf("%d", data[0]);
    for(int i=1; i<elementCount; i++) printf(" %d", data[i]);
    printf("\n");
}
void selectSort(int data[], int elementCount){
    for(int i=0; i<elementCount-1; i++){
        int minIdx = findMin(data, i, elementCount-1);
        if(minIdx != i){
            int temp = data[i];
            data[i] = data[minIdx];
            data[minIdx] = temp;
        }
        outputData(data, elementCount);
    }
}
