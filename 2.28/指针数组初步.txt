指针数组初步
题目描述
将字符串按空格分割成单词，用指针数组指向每个单词的开始位置，并将空格替换为 '0'，最后打印每个单词。

函数接口
c
int getString(char source, char strPtr[]);
裁判测试程序样例
c
#includestdio.h
int getString( char  source , char strPtr[] ) ;
int main()
{
    char str[100005];
    char strPtr[1005]={0};
    int i, num;
    gets(str);
    num = getString( str , strPtr ) ;
    for( i = 0 ; i  num ; i++ ) puts(strPtr[i]);
    return 0;    
}
输入样例
text
You are great
输出样例
text
You
are
great

错误示范，没考虑有多个空号，而且存在越界风险
int getString(char *source, char *strPtr[]){
    int i=0;
    int j=1;
    strPtr[0]=&source[0];
    while(source[i]!='\0'){
        if(source[i]==' '){
          strPtr[j]=&source[i+1];
//这里i+1有可能越界，而且也没考虑有多个空格的可能
          j++;  
          source[i]='\0';
        }
        i++;
    }
    return j;
}


正确示范
int getString(char *source, char *strPtr[]){
    int i=0;
    int j=0;
    int inword=0;
    while(source[i]!='\0'){
        if(source[i]!=' '){
            if(!inword){
                strPtr[j]=&source[i];
                j++;
                inword=1;
            }
            i++;
        }
        else {
            inword=0;
            source[i]='\0';
            i++;
        }
    }
    return j;
}
